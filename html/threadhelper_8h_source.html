47 // This is a structure that will be passed to every thread, it does not matter which function uses it,
48 // since it is meant as a kind of "universal" container for thread-related information
49 // This is really internal, the user do no initialize it. It is filled by one of the helper functions explained immediately after.
50 // Nevertheless, the user can access to it inside each thread, and he/she should do it exclusively through functions GetNumThreads
51 // and GetThisThreadNumber and the macro GetField
52 typedef struct
53 {
54  unsigned int numthreads; // How many thread we are creating for this task
55  unsigned int num_this_thread; // Which one of them is this particular thread
56  void *ioargs; // A pointer to an array of arbitrary structures used to pass inputs and return outputs from each thread.
57 } arg_to_thread;

75 // Convenience macros to get the member of a structure inside a thread to which only a pointer to the structure has been passed (because this is
76 // the only thing allowed by the pthread library)
77 // Obtain field F from a struct of type S pointed by the pointer a. S can be a templated structure since mystruct<something> is passed as an argument, in the same
78 // way it is with a non-templated structure
79 #define GetField(a,S,F) ((S *)(((arg_to_thread *)a)->ioargs))->F
